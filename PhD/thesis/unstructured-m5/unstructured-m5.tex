	\phi_p^{n-1}
	- \lambda_p \phi_{p, vnbd}^{min} \sum_{f \in \mathcal{F}_p^-} a_{pf}

	\left( \phi_p^{n,k-1} + \Psi_{f}^{} \mathcal{D}^{+,k-1}_{pf}\phi \right) a_{pf}
\end{equation*}

	\phi_{p, vnbd}^{max}

\end{equation*}
